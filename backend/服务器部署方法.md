# 基础依赖
# Ubuntu/Debian
`apt update && apt upgrade -y`

`apt install -y python3 python3-pip python3-venv sqlite3 nginx git curl`

# CentOS/RHEL
`yum update -y`

`yum install -y python3 python3-pip sqlite nginx git curl`

# 创建项目目录并上传本地代码
`mkdir -p /opt/church-schedule && cd /opt/church-schedule`

`scp -r ~/Downloads/church-schedule/backend root@47.114.75.41:/opt/church-schedule/`

`scp -r ~/Downloads/church-schedule/frontend root@47.114.75.41:/opt/church-schedule/`

# python 依赖
`cd /opt/church-schedule/backend && pip3 install -r requirements.txt`

# 后台运行
`nohup python3 app.py 8080 > /var/log/church-schedule.log 2>&1 &`
